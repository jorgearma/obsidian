```bash
# Instalar un módulo
npm install <nombre-del-módulo>

# Instalar un módulo globalmente
npm install -g <nombre-del-módulo>

# Instalar todas las dependencias de acuerdo al package.json
npm install

# Eliminar un módulo
npm uninstall <nombre-del-módulo>

# Eliminar un módulo globalmente
npm uninstall -g <nombre-del-módulo>

# Listar todas las dependencias y dependencias de desarrollo
npm list

# Instalar un módulo como dependencia de desarrollo
npm install <nombre-del-módulo> --save-dev

# Actualizar un módulo
npm update <nombre-del-módulo>

# Instalar una versión específica de un módulo
npm install <nombre-del-módulo>@<versión>

# Listar solo las dependencias de desarrollo
npm list --dev


```
